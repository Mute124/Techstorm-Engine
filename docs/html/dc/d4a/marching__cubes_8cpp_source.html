42Point MarchingCubes::interpolate(Point& v1, float val1, Point& v2, float val2, float isovalue)
43{
44 Point interpolated;
45 float mu = (isovalue - val1) / (val2 - val1);
46
47 interpolated.x = mu*(v2.x - v1.x) + v1.x;
48 interpolated.y = mu*(v2.y - v1.y) + v1.y;
49 interpolated.z = mu*(v2.z - v1.z) + v1.z;
50
51 return interpolated;
52}
